Codeforces-1034-Div.3 题解

Codeforces Round 1034 Div.3 A-G 题解，来自旧博客 p=59。

#Codeforces#Div3#题解

A

思路

Bob 赢的条件更苛刻，需要选择特定要求的数字，而 Alice 则有数字就行，考虑从 Bob 入手。Bob 想赢，首先要满足 $n$ 为奇数，其次 $n-1$ 个数满足配对要求。

void LuoTianyi(){
    int n;
    cin>>n;
    vector<int> a(n+1);
    map<int,int> mp;
    _rep(i,0,n) mp[i%4]++;
    if(abs(mp[0]-mp[3])+abs(mp[1]-mp[2])<1) cout<<"Bob"<<endl;
    else cout<<"Alice"<<endl;
}

B

选手 $j$ 有入围的可能即可。对于大于 $a_j$ 的选手，只要 $k>1$ 就可以让他们自相残杀；$k=1$ 时只有 $j$ 是最强才能赢。

void LuoTianyi(){
    int n,j,k;
    cin>>n>>j>>k;
    vector<int> a(n+1);
    rep(i,1,n) cin>>a[i];
    if(a[j]==MAX(a)||k>1) Yes;
    else No;
}

C

拿前缀看，想保住第 $i$ 个数，它必须是前 $i$ 个数中的最小值；后缀同理。维护前缀最小值和后缀最大值数组，$a_i$ 是其中一个就输出 $1$，否则 $0$。

void LuoTianyi(){
    int n;
    cin>>n;
    vector<int> a(n+1);
    rep(i,1,n) cin>>a[i];
    vector<int> pre(n+1),suf(n+1);
    pre[1]=a[1];
    rep(i,2,n) pre[i]=min(pre[i-1],a[i]);
    suf[n]=a[n];
    dep(i,n-1,1) suf[i]=max(suf[i+1],a[i]);
    rep(i,1,n){
        if(pre[i]==a[i]||suf[i]==a[i]) cout<<1;
        else cout<<0;
    }
    ENDL;
}

D

Alice 若能一次性清空则直接获胜。否则把字符串中 $1$ 的数量看作分数，问题转化成 Bob 是否能得到 $k$ 个连续的 $0$。

void LuoTianyi(){
    int n,k;
    cin>>n>>k;
    string s;
    cin>>s;
    s=' '+s;
    int cnt1=0;
    rep(i,1,n) cnt1+=(s[i]=='1');
    if(cnt1<=k) cout<<"Alice"<<endl;
    else if(k>n/2) cout<<"Alice"<<endl;
    else cout<<"Bob"<<endl;
}

E

想把 MEX 改变为 $i$，最少需要把所有的 $i$ 删除，最多删除除了 $0$ 到 $i-1$ 以外的 $n-i$ 个数。构造差分数组统计每个删除数量对应的可行 MEX 个数。

void LuoTianyi(){
    int n;
    cin>>n;
    vector<int> a(n+1);
    map<int,int> mp;
    rep(i,1,n){ cin>>a[i]; mp[a[i]]++; }
    vector<int> diff(n+2);
    rep(i,0,n){
        int L=mp[i], R=n-i;
        if(L<=R){ diff[L]++; diff[R+1]--; }
        if(mp[i]==0) break;
    }
    vector<int> ans(n+1);
    ans[0]=diff[0];
    rep(i,1,n) ans[i]=ans[i-1]+diff[i];
    rep(i,0,n) cout<<ans[i]<<' ';
    ENDL;
}

F

$1$ 和 $i>\frac n2$ 且 $i$ 为素数时一定是不可动点。其余数字按最大素因数分组，在组内向后错开一位即可。

G

把 $k$ 连续加若干次，等价于每次把元素加上 $\gcd(k,m)$。拆出 $m$ 的因数，枚举 $\gcd(k,m)$，在每次单点修改时维护总增量和差分数组即可。
